<script>

const GAS_WEB_APP_URL = "https://script.google.com/macros/s/AKfycbzTJKw-Fc0z7RRTrXwY_dBGvpiM26PmEEPxg9A6Kpy-p2k9Vo38K-GCVZFvCBH0APIv/exec"; // ここにGASのウェブアプリURLを入れる。空欄ならlocalStorage保存。

const STORAGE_KEY = "tokyo-railmap-memo-v1";


const rankMap = {

  love: { emoji: "😍", label: "とっても住みたい" },

  want: { emoji: "⭐", label: "住みたい" },

  low:  { emoji: "👀", label: "少し気になる" }

};


let markers = [];

let selectedId = null;

let placing = false;


const overlay = document.getElementById("overlay");

const placeBtn = document.getElementById("placeBtn");

const rankSelect = document.getElementById("rankSelect");

const rankEdit = document.getElementById("rankEdit");

const editPanel = document.getElementById("editPanel");

const stationInput = document.getElementById("stationInput");

const memoInput = document.getElementById("memoInput");

const saveBtn = document.getElementById("saveBtn");

const closeBtn = document.getElementById("closeBtn");

const deleteBtn = document.getElementById("deleteBtn");

const reloadBtn = document.getElementById("reloadBtn");

const statusEl = document.getElementById("status");

const filterInput = document.getElementById("filterInput");


function setStatus(text) { statusEl.textContent = text; }

function uid() { return crypto.randomUUID ? crypto.randomUUID() : String(Date.now()) + Math.random(); }


async function loadMarkers() {

  setStatus("読み込み中...");

  try {

    if (GAS_WEB_APP_URL) {

      const res = await fetch(GAS_WEB_APP_URL + "?t=" + Date.now());

      const data = await res.json();

      markers = Array.isArray(data.markers) ? data.markers : [];

    } else {

      markers = JSON.parse(localStorage.getItem(STORAGE_KEY) || "[]");

    }

    renderMarkers();

    setStatus("読み込み完了");

  } catch (e) {

    console.error(e);

    setStatus("読み込み失敗");

  }

}


async function persistMarkers() {

  setStatus("保存中...");

  try {

    if (GAS_WEB_APP_URL) {

      await fetch(GAS_WEB_APP_URL, {

        method: "POST",

        mode: "no-cors",

        headers: { "Content-Type": "text/plain;charset=utf-8" },

        body: JSON.stringify({ markers })

      });

    } else {

      localStorage.setItem(STORAGE_KEY, JSON.stringify(markers));

    }

    setStatus("保存済み " + new Date().toLocaleTimeString("ja-JP", { hour: "2-digit", minute: "2-digit" }));

  } catch (e) {

    console.error(e);

    setStatus("保存失敗");

  }

}


function renderMarkers() {

  overlay.querySelectorAll(".marker").forEach(el => el.remove());

  const q = filterInput.value.trim().toLowerCase();

  for (const m of markers) {

    if (q && !(m.station || "").toLowerCase().includes(q) && !(m.memo || "").toLowerCase().includes(q)) continue;

    const el = document.createElement("div");

    el.className = "marker";

    el.style.left = (m.x * 100) + "%";

    el.style.top = (m.y * 100) + "%";

    el.title = `${m.station || "駅名未入力"} / ${rankMap[m.rank]?.label || ""}`;

    el.dataset.id = m.id;

    el.textContent = rankMap[m.rank]?.emoji || "⭐";

    if ((m.memo || "").trim()) {

      const dot = document.createElement("span");

      dot.className = "memo-dot";

      dot.textContent = "✏️";

      el.appendChild(dot);

    }

    el.addEventListener("click", e => {

      e.stopPropagation();

      openEditor(m.id);

    });

    overlay.appendChild(el);

  }

}


function setPlacing(next) {

  placing = next;

  overlay.classList.toggle("placing", placing);

  placeBtn.classList.toggle("active", placing);

  placeBtn.textContent = placing ? "配置中：駅をクリック" : "マーク追加";

}


function openEditor(id) {

  selectedId = id;

  const m = markers.find(x => x.id === id);

  if (!m) return;

  stationInput.value = m.station || "";

  rankEdit.value = m.rank || "want";

  memoInput.value = m.memo || "";

  document.getElementById("panelTitle").textContent = (m.station || "駅名未入力") + " のメモ";

  editPanel.classList.add("show");

}


function closeEditor() {

  editPanel.classList.remove("show");

  selectedId = null;

}


placeBtn.addEventListener("click", () => setPlacing(!placing));


overlay.addEventListener("click", e => {

  if (!placing) return;

  const rect = overlay.getBoundingClientRect();

  const m = {

    id: uid(),

    station: "",

    rank: rankSelect.value,

    memo: "",

    x: (e.clientX - rect.left) / rect.width,

    y: (e.clientY - rect.top) / rect.height,

    updatedAt: new Date().toISOString()

  };

  markers.push(m);

  renderMarkers();

  setPlacing(false);

  openEditor(m.id);

});


saveBtn.addEventListener("click", async () => {

  const m = markers.find(x => x.id === selectedId);

  if (!m) return;

  m.station = stationInput.value.trim();

  m.rank = rankEdit.value;

  m.memo = memoInput.value.trim();

  m.updatedAt = new Date().toISOString();

  renderMarkers();

  await persistMarkers();

  closeEditor();

});


closeBtn.addEventListener("click", closeEditor);


deleteBtn.addEventListener("click", async () => {

  if (!selectedId) return;

  if (!confirm("このマークを削除しますか？")) return;

  markers = markers.filter(x => x.id !== selectedId);

  renderMarkers();

  await persistMarkers();

  closeEditor();

});


reloadBtn.addEventListener("click", loadMarkers);

filterInput.addEventListener("input", renderMarkers);


loadMarkers();

</script>
